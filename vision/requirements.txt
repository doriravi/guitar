# Fretboard hand-tracking prototype (Python 3.12 ONLY).
#
# Version pins matter here:
#   - MediaPipe ships no wheels for Python 3.13/3.14 — you MUST use 3.12.
#   - mediapipe 0.10.14 is the last classic release whose wheel reliably
#     includes the `mediapipe.solutions.hands` API used by this script.
#     (Some newer 0.10.x wheels shipped without the `python/solutions` tree.)
#   - opencv-python is pinned to 4.x; do NOT let opencv-contrib-python 5.x get
#     pulled in — the two share native files and shadow each other, breaking cv2.
#
# Recreate the environment:
#   py -3.12 -m venv venv
#   venv\Scripts\python -m pip install -r requirements.txt   (Windows)
#   ./venv/bin/python   -m pip install -r requirements.txt   (macOS/Linux)

mediapipe==0.10.14
opencv-python==4.10.0.84
